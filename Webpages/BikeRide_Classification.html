    NumMembers = Members['ID'].count()

        forecast_array = []
        temp = weekly_forecast[i]['temperature']
        date = weekly_forecast[i]['startTime']
        precip = weekly_forecast[i]['shortForecast']
        wind = weekly_forecast[i]["windSpeed"]
        windspeeds = [int(s) for s in wind.split() if s.isdigit()]
        windspeed = sum(windspeeds)/len(windspeeds)
        if precip.find('Shower')!= -1 or precip.find('Rain') != -1:
            prcp = 1
